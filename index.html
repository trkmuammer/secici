<script>

    function pickRandomName() {

        const textArea = document.getElementById('nameList');

        const resultDiv = document.getElementById('result');

        

        // Satırları ayır ve boşlukları temizle

        const names = textArea.value.split('\n').filter(name => name.trim() !== "");


        if (names.length === 0) {

            resultDiv.innerText = "Lütfen önce isim yazın!";

            resultDiv.style.color = "#dc3545";

            return;

        }


        // Rastgele seçim işlemi

        const randomIndex = Math.floor(Math.random() * names.length);

        const selectedName = names[randomIndex];


        // Sonucu ekrana yazdır

        resultDiv.style.color = "#28a745";

        resultDiv.innerText = "🎉 Seçilen: " + selectedName;

    }

</script>
